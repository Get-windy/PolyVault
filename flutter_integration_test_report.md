testWidgets('凭证完整生命周期', (WidgetTester tester) async {
  // 1. 启动应用
  await tester.pumpWidget(const PolyVaultApp());
  await tester.pumpAndSettle();
  
  // 2. 导航到凭证页面
  await tester.tap(find.text('凭证管理'));
  await tester.pumpAndSettle();
  
  // 3. 添加凭证
  await tester.tap(find.byType(FloatingActionButton));
  await tester.pumpAndSettle();
  
  await tester.enterText(
    find.byKey(const Key('serviceNameField')), 
    'TestService'
  );
  await tester.enterText(
    find.byKey(const Key('usernameField')), 
    'testuser'
  );
  await tester.enterText(
    find.byKey(const Key('passwordField')), 
    'testpass123'
  );
  
  await tester.tap(find.text('保存'));
  await tester.pumpAndSettle();
  
  // 4. 验证凭证显示
  expect(find.text('TestService'), findsOneWidget);
  expect(find.text('testuser'), findsOneWidget);
  
  // 5. 删除凭证
  await tester.tap(find.byIcon(Icons.delete));
  await tester.pumpAndSettle();
  
  await tester.tap(find.text('确认'));
  await tester.pumpAndSettle();
  
  // 6. 验证删除成功
  expect(find.text('TestService'), findsNothing);
});

**改进建议**:
```dart
class NetworkExceptionHandler {
  static Future<T> handle<T>(Future<T> operation) async {
    try {
      return await operation;
    } on SocketException catch (e) {
      // 显示网络错误
      showNetworkErrorDialog();
      // 启用离线模式
      enableOfflineMode();
      throw NetworkException('网络连接失败');
    } on TimeoutException catch (e) {
      showTimeoutError();
      throw TimeoutException('请求超时');
    }
  }
}
```
